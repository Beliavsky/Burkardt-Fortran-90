# fxyz1_x_commands.txt

# Usage:
#  gnuplot < fxyz1_x_commands.txt
#
set term png
set output "fxyz1_x.png"
set view 20, 75
set hidden3d
set timestamp
set xlabel "<--- Y --->"
set ylabel "<--- Z --->"
set zlabel "<--- X --->"
set title "Slice X = 0.0"
set grid
set style data lines
splot "fxyz1_x_data.txt" with lines
quit
